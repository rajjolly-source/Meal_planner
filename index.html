async function fetchRecipes() { const response = await fetch(csvUrl); const data = await response.text(); const lines = data.split('\n'); const headers = lines[0].toLowerCase().split(','); const idxName = headers.indexOf("dish name"); const idxCook = headers.indexOf("who's the cook?"); const idxRating = headers.indexOf("rating"); recipePool = lines.slice(1).filter(l => l.trim()).map(line => { const v = line.split(','); return { name: v[idxName], cook: v[idxCook] || "Anyone", rating: v[idxRating] || "" }; }); // Generate a fresh 14-day schedule ONLY if it's currently empty if (schedule.length === 0) { schedule = Array(14).fill(null).map(() => ({ ...recipePool[Math.floor(Math.random() * recipePool.length)], locked: false, status: 'home' })); } render(); } function initializeSchedule() { // This button now ONLY rerolls unlocked items schedule = schedule.map((item) => { if (item.locked) { return item; // Keep the locked one } else { return { ...recipePool[Math.floor(Math.random() * recipePool.length)], locked: false, status: 'home' }; } }); render(); }
